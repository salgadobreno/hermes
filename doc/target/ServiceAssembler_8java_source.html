49 byte[] data = new byte[0];

51 byte[] header, timestamp, message_content, pin;

54 if (service instanceof PasswordOptional || service instanceof RangedTimestampAble) {
55 header = headerPolicy.getHeader(service, ServiceCodeLookup.getServiceCode(
56 service,
57 service instanceof PasswordOptional && ((PasswordOptional) service).hasPin(),
58 service instanceof RangedTimestampAble && ((RangedTimestampAble) service).isRanged()));

60 header = headerPolicy.getHeader(service, null);
61 }
62 timestamp = timestampPolicy.get();
63 pin = passwordPolicy.get();
64
65 // NOTE: O Token não lê QR Nível 1, então estamos verificando isso aqui e colocando um padding
66 // junto da mensagem pois o PIN fica sempre no fim do conteúdo
67 int total_lengh = header.length + timestamp.length + message_content.length + pin.length;
68 if (total_lengh < 18) {
69 byte[] padding = new byte[18 - total_lengh];
70 for (int i = 0; i < padding.length; i++) {
71 padding[i] = 'x'; // NOTE: padding com 'x' pq vazio deu um padrão que confundia o leitor
72 }
73 message_content = addAll(message_content, padding);
74 }
75
76 //1- header
77 data = addAll(data, header);
78 //2- timestamp
79 data = addAll(data, timestamp);
80 //3- message cifrado ou n
81 data = addAll(data, message_content);
82 //4- hmac
83 data = hmacKeyPolicy.apply(data);
84 //5- password
85 data = addAll(data, pin);
