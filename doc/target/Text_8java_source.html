30 static int calcAlignment(Text textObj) {
31 int maxTextLength = 0;
32 String[] lines = textObj.getText().split("\\n");
33 for (String line : lines) {
34 if (maxTextLength < line.length()) maxTextLength = line.length();
35 }
36
37 switch (textObj.getAlignment()) {
38 case CENTER:
39 return (Token.DISPLAY_WIDTH - (maxTextLength * textObj.getSize().getWidth())) >> 1;
40 case LEFT:
41 return Token.HORIZONTAL_MARGIN;
42 case RIGHT:
43 return (Token.DISPLAY_WIDTH - (textObj.getSize().getWidth() * maxTextLength));
44 case ARGUMENT:
45 return Token.HORIZONTAL_MARGIN;
46 default:
47 throw new IllegalArgumentException("Unexpected alignment");
48 }
49 }
