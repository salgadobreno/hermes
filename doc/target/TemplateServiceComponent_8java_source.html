142 private void parseTemplateObj(TemplateObj templateObj) {
143 if (templateObj instanceof Text) {
144 Text text = (Text) templateObj;
145 if (text.getColor().isArg()) addColorArg(templateObj);
146 if (text.getBgColor().isArg()) addColorArg(templateObj);
147 if (text.getSize().isArg()) addSizeArg(templateObj);
148 if (text.getAlignment().isArg()) addAlignmentArg(templateObj);
149 if (text.isArg()) addTextArg(templateObj);
150 return;
151 }
152
153 if (templateObj instanceof Header) {
154 Header header = (Header) templateObj;
155 if (header.getBgColor().isArg()) addColorArg(templateObj);
156 if (header.getTextColor().isArg()) addColorArg(templateObj);
157 if (header.getText().equals(Text.TEXT_FROM_ARGUMENT)) addTextArg(templateObj);
158 return;
159 }
160
161 if (templateObj instanceof Footer) {
162 Footer footer = (Footer) templateObj;
163 if (footer.getBgColor().isArg()) addColorArg(templateObj);
164 if (footer.getTextColor().isArg()) addColorArg(templateObj);
165 if (footer.getText().equals(Text.TEXT_FROM_ARGUMENT)) addTextArg(templateObj);
166 if (footer.getText2().equals(Text.TEXT_FROM_ARGUMENT)) addTextArg(templateObj);
167 return;
168 }
169
170 if (templateObj instanceof Stripe) {
171 Stripe stripe = (Stripe) templateObj;
172 if (stripe.getColor().isArg()) addColorArg(templateObj);
173 return;
174 }
175
176 if (templateObj instanceof Rect) {
177 Rect rect = (Rect) templateObj;
178 if (rect.getColor().isArg()) addColorArg(templateObj);
179 }
180 }

235 @Override
236 public Service getService() throws Exception {
237 service.setTemplateSlot(templateSlotSelect.getValue());
238 service.setAesKey(ClientKeyConfiguration.getSelected().getAesKey(serialNumberField.getText()));
239 service.setHmacKey(ClientKeyConfiguration.getSelected().getHmacKey(serialNumberField.getText()));
240
241 service.setTimestampRange(rangedTimestampField.getStartDateValue(), rangedTimestampField.getEndDateValue());
242 service.togglePasswordOptional(optionalPasswordField.isOptional());
243 service.setPin(optionalPasswordField.getText());
244 service.setParams(new ArrayList<>());
245 for (Control control : controlList) {
246 if (control instanceof TextField) {
247 service.getParams().add(new HuffmanEncodedParam(((TextField)control).getText()));
248 } else if (control instanceof TemplateColorPicker) {
249 service.getParams().add(new TemplateColorParam(((TemplateColorPicker) control).getValue()));
250 } else if (control instanceof TextAlignmentSelect) {
251 service.getParams().add(new TextAlignmentParam(((TextAlignmentSelect)control).getValue()));
252 } else if (control instanceof TextSizeSelect) {
253 service.getParams().add(new TextSizeParam(((TextSizeSelect)control).getValue()));
254 }
255 }
